# 1. CRÉATION DE L'ENVIRONNEMENT VIRTUEL
# Dans le terminal, à la racine du projet MYEMO :
python3 -m venv .venv

# 2. ACTIVATION DE L'ENVIRONNEMENT VIRTUEL
# Sur macOS (votre système) :
source .venv/bin/activate

# 3. INSTALLATION DES DÉPENDANCES
pip install pandas==2.1.1
pip install espn_api==0.14.0
pip install schedule==1.2.0

# 4. VÉRIFICATION DE L'INSTALLATION
python3 -c "import pandas; import espn_api; import schedule; print('Installation réussie!')"

# 5. CRÉATION DES DOSSIERS NÉCESSAIRES (si pas déjà fait)
mkdir -p src/{collectors,processors,utils}
mkdir -p data/{raw,processed}/{standings,rosters,free_agents}
mkdir -p logs
mkdir -p scripts
mkdir -p config

# 6. DÉSACTIVATION DE L'ENVIRONNEMENT (quand vous avez fini)
deactivate
