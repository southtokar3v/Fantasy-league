# Nettoyer les anciennes données

## Supprimer les fichiers de plus de 30 jours
find data/raw -type f -mtime +30 -exec rm {} \;

## Archiver les anciens logs
mkdir -p logs/archives
mv logs/espn_collector_*.log logs/archives/

## Nettoyer les fichiers temporaires
rm -rf **/__pycache__
rm -rf .pytest_cache

## Compresser les anciennes données
tar -czf data/archives/$(date +%Y%m).tar.gz data/raw/*/$(date +%Y%m)*
